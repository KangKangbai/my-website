// 两数相加（链表）
ListNode addTwoNumbers(ListNode a, ListNode b){
    ListNode d = new ListNode(-1), p = d; int c = 0;
    while(a != null || b != null){
        int x = (a==null?0:a.val), y = (b==null?0:b.val), s = x + y + c;
        p.next = new ListNode(s%10); c = s/10; p = p.next;
        if(a!=null) a=a.next; if(b!=null) b=b.next;
    }
    if(c>0) p.next = new ListNode(c);
    return d.next;
}
